CLASS SPECIES
    ATTRIBUTES
        PRIVATE species_name: TEXT
        PRIVATE care_requirements: TEXT
        PRIVATE safe_temperature_min: NUMBER
        PRIVATE safe_temperature_max: NUMBER
        PRIVATE safe_heart_rate_min: NUMBER
        PRIVATE safe_heart_rate_max: NUMBER
    CONSTRUCTOR(speciesName: TEXT, careReq: TEXT, tempMin: NUMBER, tempMax: NUMBER, heartMin: NUMBER, heartMax: NUMBER)
        SET species_name TO speciesName
        SET care_requirements TO careReq
        SET safe_temperature_min TO tempMin
        SET safe_temperature_max TO tempMax
        SET safe_heart_rate_min TO heartMin
        SET safe_heart_rate_max TO heartMax
    END CONSTRUCTOR
    METHODS
        CHECK_TEMPERATURE(temp: NUMBER)
            IF temp LESS THAN safe_temperature_min OR temp GREATER THAN safe_temperature_max THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD
        CHECK_HEART_RATE(rate: NUMBER)
            IF rate LESS THAN safe_heart_rate_min OR rate GREATER THAN safe_heart_rate_max THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD
        GET_SPECIES_NAME()
            RETURN species_name
        END METHOD
        GET_CARE_REQUIREMENTS()
            RETURN care_requirements
        END METHOD
END CLASS

CLASS EXAMINATION
    ATTRIBUTES
        PRIVATE diagnosis: TEXT
        PRIVATE temperature: NUMBER
        PRIVATE heart_rate: NUMBER
        PRIVATE treatments: LIST<TREATMENT>[]
        PRIVATE urgent: BOOLEAN
    CONSTRUCTOR(examDiagnosis: TEXT, temp: NUMBER, heartRate: NUMBER)
        SET diagnosis TO examDiagnosis
        SET temperature TO temp
        SET heart_rate TO heartRate
        SET treatments TO EMPTY LIST
        SET urgent TO FALSE
    END CONSTRUCTOR
    METHODS
        ADD_TREATMENT(newTreatment: TREATMENT)
            ADD newTreatment TO treatments
            DISPLAY "TREATMENT ADDED"
        END METHOD
        CHECK_URGENCY(petSpecies: SPECIES)
            IF petSpecies.CHECK_TEMPERATURE(temperature) EQUALS TRUE OR petSpecies.CHECK_HEART_RATE(heart_rate) EQUALS TRUE THEN
                SET urgent TO TRUE
                DISPLAY "URGENT EXAMINATION"
            ELSE
                SET urgent TO FALSE
                DISPLAY "NORMAL EXAMINATION"
            END IF
        END METHOD
        GET_URGENCY()
            RETURN urgent
        END METHOD
        CALCULATE_TREATMENT_TOTAL()
            SET total TO 0
            REPEAT FROM I = 0 TO SIZE OF treatments - 1
                SET total TO total + treatments[I].GET_COST()
            END REPEAT
            RETURN total
        END METHOD
END CLASS

        END METHOD

CLASS SCREEN
    ATTRIBUTES
        PRIVATE screen_id: NUMBER
        PRIVATE screen_name: TEXT
        PRIVATE capacity: NUMBER
        PRIVATE seats: LIST<TEXT>[]
        PRIVATE showtimes: LIST<SHOWTIME>[]
    CONSTRUCTOR(screenId: NUMBER, screenName: TEXT, screenCapacity: NUMBER, seatList: LIST<TEXT>)
        SET screen_id TO screenId
        SET screen_name TO screenName
        SET capacity TO screenCapacity
        SET seats TO seatList
        SET showtimes TO EMPTY LIST
    END CONSTRUCTOR
    METHODS
        GET_CAPACITY()
            RETURN capacity
        END METHOD
        GET_SEATS()
            RETURN seats
        END METHOD
        ADD_SHOWTIME(newShowtime: SHOWTIME)
            REPEAT FROM I = 0 TO SIZE OF showtimes - 1
                IF newShowtime.CHECK_OVERLAP(showtimes[I]) EQUALS TRUE THEN
                    DISPLAY "SHOWTIME OVERLAPS WITH EXISTING SHOWTIME"
                    RETURN FALSE
                END IF
            END REPEAT
            ADD newShowtime TO showtimes
            DISPLAY "SHOWTIME ADDED SUCCESSFULLY"
            RETURN TRUE
        END METHOD
END CLASS
CLASS SHOWTIME
    ATTRIBUTES
        PRIVATE showtime_id: NUMBER
        PRIVATE film: FILM
        PRIVATE screen: SCREEN
        PRIVATE start_time: NUMBER
        PRIVATE end_time: NUMBER
        PRIVATE booked_seats: LIST<TEXT>[]
    CONSTRUCTOR(showtimeId: NUMBER, showFilm: FILM, showScreen: SCREEN, startTime: NUMBER, endTime: NUMBER)
        SET showtime_id TO showtimeId
        SET film TO showFilm
        SET screen TO showScreen
        SET start_time TO startTime
        SET end_time TO endTime
        SET booked_seats TO EMPTY LIST
    END CONSTRUCTOR
    METHODS
        CHECK_OVERLAP(otherShowtime: SHOWTIME)
            IF start_time LESS THAN otherShowtime.end_time AND end_time GREATER THAN otherShowtime.start_time THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD
        BOOK_SEAT(seatNumber: TEXT)
            IF seatNumber NOT IN screen.GET_SEATS() THEN
                DISPLAY "SEAT DOES NOT EXIST"
                RETURN FALSE
            END IF
            IF seatNumber IN booked_seats THEN
                DISPLAY "SEAT ALREADY BOOKED"
                RETURN FALSE
            ELSE
                ADD seatNumber TO booked_seats
                DISPLAY "SEAT BOOKED SUCCESSFULLY"
                RETURN TRUE
            END IF
        END METHOD
        GET_AVAILABLE_SEATS()
            RETURN screen.GET_CAPACITY() - SIZE OF booked_seats
        END METHOD
        GET_FILM()
            RETURN film
        END METHOD
        GET_SCREEN()
            RETURN screen
        END METHOD
END CLASS
